<script>
(function () {

  /* ============================================================
    CONFIGURATION – edit this array to customize the wheel.

    Each object represents one segment:
      lns    {string[]}  Label lines shown on / near the segment.
                         Use two strings for a two-line label.
      desc   {string}    Center description text. Use \n for line breaks.
                         Keep to ~3 short lines for best fit.
      col    {string}    Solid hex color when segment is active.
      bg     {string}    Light hex tint when segment is inactive.
      ib     {string}    Icon circle background color (usually same as col).

    Segments are placed clockwise starting from 12 o'clock.
  ============================================================ */
  var values = [
    {
      lns:  ['Integrity'],
      desc: 'We act with honesty,\nrespect, and care\nin all our endeavors.',
      col:  '#8fa882',
      bg:   '#e0ebe0',
      ib:   '#8fa882'
    },
    {
      lns:  ['Innovative', 'Thinking'],
      desc: 'We embrace open-mindedness\nand creativity in pursuit\nof every opportunity.',
      col:  '#6a9ec4',
      bg:   '#d8eaf4',
      ib:   '#6a9ec4'
    },
    {
      lns:  ['Community', 'Impact'],
      desc: 'We work to positively impact\nthe communities where\nwe invest and live.',
      col:  '#2d4e70',
      bg:   '#d4e1ee',
      ib:   '#2d4e70'
    },
    {
      lns:  ['Collaboration'],
      desc: 'We promote an inclusive\nculture built on trust\nand a shared vision.',
      col:  '#364f68',
      bg:   '#d4e1ee',
      ib:   '#364f68'
    },
    {
      lns:  ['Relationships'],
      desc: 'We build enduring relationships\nwith our investors and\noperating partners.',
      col:  '#4b5f6f',
      bg:   '#d7e3ec',
      ib:   '#4b5f6f'
    }
  ];

  /* ── Wheel geometry constants ── */
  var CX     = 320;   /* SVG center X */
  var CY     = 300;   /* SVG center Y */
  var RI     = 113;   /* inner (donut hole) radius */
  var RD     = 193;   /* outer radius when inactive */
  var RA     = 263;   /* outer radius when active (expands past label) */
  var RLBL   = 250;   /* radial distance of label text */
  var RICO   = 172;   /* radial distance of icon circle center */
  var RICR   = 19;    /* icon circle radius */
  var GAP    = 0.038; /* gap between segments in radians */
  var N      = values.length;
  var SA     = (Math.PI * 2) / N;          /* angle per segment */
  var S0     = -Math.PI / 2 - SA / 2;     /* start so seg 0 is centred at top */
  var NS     = 'http://www.w3.org/2000/svg';

  var active = -1; /* index of active segment, -1 = none */

  /* ── Helpers ── */
  function polar(r, a) {
    return [CX + r * Math.cos(a), CY + r * Math.sin(a)];
  }
  function f(n) { return n.toFixed(2); }

  function buildPath(i, isActive) {
    var ro = isActive ? RA : RD;
    var a0 = S0 + i * SA + GAP;
    var a1 = S0 + (i + 1) * SA - GAP;
    var lg = (a1 - a0 > Math.PI) ? 1 : 0;
    var ix0 = polar(RI, a0), ox0 = polar(ro, a0);
    var ox1 = polar(ro, a1), ix1 = polar(RI, a1);
    return [
      'M', f(ix0[0]), ',', f(ix0[1]),
      'L', f(ox0[0]), ',', f(ox0[1]),
      'A', ro, ',', ro, ' 0 ', lg, ',1 ', f(ox1[0]), ',', f(ox1[1]),
      'L', f(ix1[0]), ',', f(ix1[1]),
      'A', RI, ',', RI, ' 0 ', lg, ',0 ', f(ix0[0]), ',', f(ix0[1]), 'Z'
    ].join('');
  }

  function svgEl(tag, attrs) {
    var el = document.createElementNS(NS, tag);
    Object.keys(attrs).forEach(function (k) { el.setAttribute(k, attrs[k]); });
    return el;
  }

  /* ── Main render ── */
  function render() {
    var pathsG  = document.getElementById('mw-paths');
    var labelsG = document.getElementById('mw-labels');
    var iconsG  = document.getElementById('mw-icons');
    pathsG.innerHTML  = '';
    labelsG.innerHTML = '';
    iconsG.innerHTML  = '';

    values.forEach(function (v, i) {
      var isAct = (i === active);
      var mid   = S0 + (i + 0.5) * SA;
      var lPos  = polar(RLBL, mid);
      var iPos  = polar(RICO, mid);
      var cos   = Math.cos(mid);
      var anc   = cos > 0.28 ? 'start' : cos < -0.28 ? 'end' : 'middle';

      /* — Segment — */
      var segG = svgEl('g', { class: 'mw-seg-btn', role: 'button', tabindex: '0', 'aria-label': v.lns.join(' ') });
      var path = svgEl('path', {
        d:             buildPath(i, isAct),
        fill:          isAct ? v.col : v.bg,
        stroke:        'var(--mw-seg-stroke)',
        'stroke-width':    'var(--mw-seg-stroke-w)',
        'stroke-linejoin': 'round'
      });
      segG.appendChild(path);
      segG.addEventListener('click',   function () { toggle(i); });
      segG.addEventListener('keydown', function (e) { if (e.key === 'Enter' || e.key === ' ') toggle(i); });
      pathsG.appendChild(segG);

      /* — Label — */
      var labelG = svgEl('g', { 'pointer-events': 'none' });
      var nLines = v.lns.length;
      var baseY  = lPos[1] - (nLines - 1) * 8.5;
      v.lns.forEach(function (ln, li) {
        var t = svgEl('text', {
          x:             f(lPos[0]),
          y:             f(baseY + li * 17),
          'text-anchor': anc,
          'font-size':   '13',
          'font-weight': '500',
          'font-family': 'var(--mw-font)',
          fill:          isAct ? '#ffffff' : 'var(--mw-text)'
        });
        t.textContent = ln;
        labelG.appendChild(t);
      });
      labelsG.appendChild(labelG);

      /* — Icon circle — */
      var iconG = svgEl('g', { class: 'mw-seg-btn' });
      iconG.addEventListener('click', function () { toggle(i); });
      var circ = svgEl('circle', {
        cx:           f(iPos[0]),
        cy:           f(iPos[1]),
        r:            RICR,
        fill:         isAct ? '#ffffff' : v.ib,
        stroke:       isAct ? v.col : 'none',
        'stroke-width': '1.5'
      });
      var sign = svgEl('text', {
        x:                  f(iPos[0]),
        y:                  f(iPos[1]),
        'text-anchor':      'middle',
        'dominant-baseline':'central',
        'font-size':        '21',
        'font-weight':      '300',
        'font-family':      'var(--mw-font)',
        fill:               isAct ? v.col : '#ffffff',
        'pointer-events':   'none'
      });
      sign.textContent = isAct ? '\u00d7' : '+';
      iconG.appendChild(circ);
      iconG.appendChild(sign);
      iconsG.appendChild(iconG);
    });

    /* — Center content — */
    var logo   = document.getElementById('mw-logo');
    var descG  = document.getElementById('mw-desc');
    var dt     = document.getElementById('mw-dt');

    if (active === -1) {
      logo.setAttribute('opacity', '1');
      descG.setAttribute('opacity', '0');
    } else {
      logo.setAttribute('opacity', '0');
      descG.setAttribute('opacity', '1');
      dt.innerHTML = '';
      var lines   = values[active].desc.split('\n');
      var startY  = CY - ((lines.length - 1) * 18) / 2;
      dt.setAttribute('y', f(startY));
      lines.forEach(function (ln, li) {
        var ts = document.createElementNS(NS, 'tspan');
        ts.textContent = ln;
        ts.setAttribute('x', '320');
        ts.setAttribute('dy', li === 0 ? '0' : '18');
        dt.appendChild(ts);
      });
    }
  }

  function toggle(i) {
    active = (active === i) ? -1 : i;
    render();
  }

  render();

})();
</script>
